bool result = function findExit() 
begin
	for i = 0:2 beginfor
	right #смотрим направо и проверяем есть ли там стена
	dist <- wall
	if dist < 1 beginif 	#то есть расстояние до стены меньше 1, подразуемается что 0
		left #вращаемся обратно влево
		dist2 <- wall
		if dist2 > 0 beginif #если там нет стены нужно сделать шаг вперед
			move(wall)
		endif
		if dist2 < 1 beginif #если же там стена то поворачиваемся налево еще раз
			left
		endif
	endif
	if dist > 0 beginif #если нет справа стены то мы должны идти туда и следовать правую стену
		move(wall)
	endif
endfor
end
